# 模式匹配

## match

```scala
package com.stanlong.scala

/**
 * 模式匹配
 * 要匹配时，会从第一个 case 分支开始， 匹配成功则会执行对应的逻辑代码。
 * 匹配不成功，继续进行下一个分支判断。
 * 如果所有case都不匹配，那么会执行case _ 分支
 * 没有匹配到结果又没有写case _ 分支，那么会抛出MatchError
 */
object HelloScala {

    def main(args: Array[String]): Unit = {

        val n1 = 20
        val n2 = 10
        def matchDualOp(op: Char): Any  = op match {
            case '+' => n1 + n2
            case '-' => n1 - n2
            case '*' => n1 * n2
            case '/' => n1 / n2
            case _ => "非法运算符"
        }
        println(matchDualOp('+'))
        println(matchDualOp('-'))
        println(matchDualOp('*'))
        println(matchDualOp('/'))
        println(matchDualOp('$'))

    }
}
```

## 模式守卫

```scala
package com.stanlong.scala

/**
 * 模式守卫
 */
object HelloScala {

    def main(args: Array[String]): Unit = {

        // 求一个整数的绝对值
        def abs(num: Int): Int ={
            num match{
                case i if i>=0 =>i // 匹配一个范围
                case i if i<0 => -i
            }
        }
        println(abs(10))
        println(abs(-10))
    }
}
```

## 类型匹配

```scala
package com.stanlong.scala

/**
 * 类型匹配
 * 1. 匹配常量
 * 2. 匹配类型
 * 3. 匹配数组
 * 4. 匹配列表
 * 5. 匹配元组
 */
object HelloScala {

    def main(args: Array[String]): Unit = {
        // 1. 匹配常量
        def describeConst(x:Any):String = x match {
            case 1 => "Int one"
            case "hello" => "String hello"
            case true => "Boolean true"
            case '+' => "Char +"
            case _ => ""
        }

        // 2. 匹配类型
        def describeType(x: Any):String = x match {
            case i: Int => "Int" + i
            case s: String => "String" + s
            case list: List[String] => "List" + list
            case array:Array[Int] => "Array[Int]" + array.mkString(",")
            case a => "Something else: " + a
        }

        // 3. 匹配数组
        for(arr <- List(
            Array(0),
            Array(1, 0),
            Array(0,1,0),
            Array(1,1,0),
            Array(2,3,7,15),
            Array("hello",20, 30)
        )){
            val result = arr match {
                case Array(0) => "0"
                case Array(1, 0) => "Array(1, 0)"
                case Array(x, y) =>"Array:" + x +" , " + y // 匹配两个元素的数组
                case Array(0, _*) => "以0开头的数组"
                case Array(x,1,z) => "中间为1的三元素数组"
                case _ => "Something else"
            }
            println(result)
        }

        // 4.匹配列表
        for(list <- List(
            List(0),
            List(1,0),
            List(0,0,0),
            List(1,1,0),
            List(88)
        )){
            val result = list match {
                case List(0) => "0"
                case List(x, y) => "List(x, y): " + x + " , " + y
                case List(0, _*) => "List(0, ...)"
                case List(a) => "List(a): " + a
                case _ => "Something else"
            }
            println(result)
        }

        //4. 匹配列表： 方式二
        val list = List(1,2,3,4,5,67)
        list match {
            case first :: second :: rest => println(s"first: $first, second: $second, rest: $rest") // first: 1, second: 2, rest: List(3, 4, 5, 67)
            case _ => println("something else")
        }

        // 5. 匹配元组
        for(tuple <- List(
            (0,1),
            (0,0),
            (0,1,0),
            (0,1,1),
            (1,23,56),
            ("hello", true, 0.5)
        )){
            val result = tuple match {
                case (a, b) => "" + a + "," + b
                case (0,_) => ("0, _")
                case (a, 1, _) => "(a, 1, _) " + a
                case (x, y, z) => "(x, y, z) " + x + " " + "y" + " " + z
                case _ => "something else"
            }
            println(result)
        }

        // 5. 元组匹配扩展
        // 在变量声明时匹配
        val (x, y) = (10, "hello")
        println(s"x: $x, y: $y")

        val List(first, second, _*) = List(12, 13, 14, 15)
        println(s"first $first, second: $second")

        val fir :: sec :: rest = List(12, 13, 14, 15)
        println(s"fir : $fir, second: $sec, rest: $rest")

        // for 推导式中进行模式匹配
        val list1 = List(("a", 1),("b", 2),("c", 3),("d", 4),("e", 5))
        // 1. 正常遍历
        for(elem <- list1){
            println(elem._1 + " " + elem._2)
        }

        // 2. 将list的元素直接定义成元组，对遍历赋值
        for((word, count) <- list1){
            println(word + ": " + count)
        }

        // 3. 可以不考虑某个位置的变量，只变量key或者value
        for((word, _) <- list1){
            println(word)
        }

        // 4. 可以指定某个位置的值必须是多少
        for(("a", count) <- list1){ // 相当于直把key="a" 的元组打印出来
            println(count)
        }
    }
}
```

## 匹配对象

```scala
package com.stanlong.scala

/**
 * 对象匹配
 */
object HelloScala {

    def main(args: Array[String]): Unit = {

        val student = new Student("zhangsan", 18)

        // 针对对象实例的内容进行匹配
        val result = student match {
            case Student("zhangsan", 18) => "Alice, 18"
            case _ => "Else"
        }
        println(result)
    }
}

// 要匹配的类
class Student(val name: String, val age: Int)

// 如果针对对象实例的内容进行匹配则需要定义伴生对象
object Student{
    def apply(name: String, age: Int): Student = new Student(name, age)

    // 且必须实现 unapply 方法，用来对对象属性进行拆解
    def unapply(student: Student): Option[(String, Int)] = {
        if(student == null){
            None
        }else{
            Some(student.name, student.age)
        }
    }
}
```

## 样例类

```scala
package com.stanlong.scala

/**
 * 对象匹配
 * 针对对象实例的内容进行匹配使用样例类是最为简单快捷的， 不需要定义伴生对象并实行相关的方法
 */
object HelloScala {

    def main(args: Array[String]): Unit = {

        val student = new Student("zhangsan", 18)

        // 针对对象实例的内容进行匹配
        val result = student match {
            case Student("zhangsan", 18) => "Alice, 18"
            case _ => "Else"
        }
        println(result)
    }
}

// 定义样例类， 针对对象实例的内容进行匹配使用样例类是最为简单快捷的
case class Student(val name: String, val age: Int)
```
