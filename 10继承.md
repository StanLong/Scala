# 继承

```scala
package com.stanlong.scala
/**
 * 继承
 */
object Exercise01 extends App { // 继承App后，就可以直接在这个类中执行代码，不需要再写main入口
    println("app~~")
}
```

```scala
package com.stanlong.scala

/**
 * 子类继承父类
 */
object Exercise01{
    def main(args: Array[String]): Unit = {
        val student = new Student
        student.name = "StanLong"
        student.age = 27
        student.studying()
        student.showInfo()
    }
}


class Person(){
    var name : String = _
    var age : Int = _

    def showInfo(): Unit = {
        println("学生信息如下")
        println("名字: " + this.name)
        println("年龄: " + this.age)
    }
}

class Student extends Person{
    def studying(): Unit = {
        println(this.name + " 学习scala...")
    }
}
```

## 方法重写

> def只能重写另一个def(即：方法只能重写另一个方法)

```scala
package com.stanlong.scala

/**
 * 方法重写
 */
object Exercise01{ // 继承App后，就可以直接在这个类中执行代码，不需要再写main入口
    def main(args: Array[String]): Unit = {
        val student = new Student
        student.name = "StanLong"
        student.age = 27
        student.studying()
        student.showInfo()
    }
}


class Person(){
    var name : String = _
    var age : Int = _

    def showInfo(): Unit = {
        println("学生信息如下")
        println("名字: " + this.name)
        println("年龄: " + this.age)
    }
}

class Student extends Person{
    def studying(): Unit = {
        println(this.name + " 学习scala...")
    }

    // 需要显示的使用 override 关键字
    override def showInfo(): Unit = {
        println("方法重写")
        super.showInfo()
    }
}
```

## 属性重写

> val只能重写另一个val 属性 或 重写不带参数的def

```scala
package com.stanlong.scala

/**
 * 属性重写
 */
object Exercise01 {
    def main(args: Array[String]): Unit = {
        val person = new Person
        val student = new Student
        println(person.name) // StanLong
        println(student.name) // 南海财神

        val person2 = new Person
        val student2 = new Person
        println(person2.name) // StanLong
        println(student2.name) // StanLong

        val person3 = new Student
        val student3 = new Student
        println(person3.name) // 南海财神
        println(student3.name) // 南海财神
    }
}

class Person{
    val name:String = "StanLong"
}

class Student extends Person{
    override val name:String = "南海财神"
}
```
