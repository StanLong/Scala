# 继承

```scala
package com.stanlong.scala
/**
 * 继承
 */
object Exercise01 extends App { // 继承App后，就可以直接在这个类中执行代码，不需要再写main入口
    println("app~~")
}
```

```scala
package com.stanlong.scala

/**
 * 子类继承父类
 */
object Exercise01{
    def main(args: Array[String]): Unit = {
        val student = new Student
        student.name = "StanLong"
        student.age = 27
        student.studying()
        student.showInfo()
    }
}


class Person(){
    var name : String = _
    var age : Int = _

    def showInfo(): Unit = {
        println("学生信息如下")
        println("名字: " + this.name)
        println("年龄: " + this.age)
    }
}

class Student extends Person{
    def studying(): Unit = {
        println(this.name + " 学习scala...")
    }
}
```

## 方法重写

> def只能重写另一个def(即：方法只能重写另一个方法)

```scala
package com.stanlong.scala

/**
 * 方法重写
 */
object Exercise01{ // 继承App后，就可以直接在这个类中执行代码，不需要再写main入口
    def main(args: Array[String]): Unit = {
        val student = new Student
        student.name = "StanLong"
        student.age = 27
        student.studying()
        student.showInfo()
    }
}


class Person(){
    var name : String = _
    var age : Int = _

    def showInfo(): Unit = {
        println("学生信息如下")
        println("名字: " + this.name)
        println("年龄: " + this.age)
    }
}

class Student extends Person{
    def studying(): Unit = {
        println(this.name + " 学习scala...")
    }

    // 需要显示的使用 override 关键字
    override def showInfo(): Unit = {
        println("方法重写")
        super.showInfo()
    }
}
```

## 属性重写

> val只能重写另一个val 属性 或 重写不带参数的def

```scala
package com.stanlong.scala

import scala.reflect.internal.FatalError

/**
 * 属性重写
 */
object Exercise01 {
    def main(args: Array[String]): Unit = {
        var father = new Father
        father.fatherName() // StanLong

        val son = new Son
        son.sonName()    // 南海财神
        son.fatherName() // 南海财神   子类覆写父类name属性
    }
}

class Father{
    val name:String = "StanLong"

    def fatherName(): Unit ={
        println(name)
    }
}

class Son extends Father{
    override val name:String = "南海财神" // 子类覆写父类name属性

    def sonName(): Unit ={
        println(name)
    }
}
```
